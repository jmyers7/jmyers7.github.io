import numpy as np
import plotly.graph_objects as go

# Define the number of terms to plot
n_terms = 10

# Define x values
x = np.linspace(-np.pi, np.pi, 1000)

# Example: Square wave Fourier series
# Initialize the figure
fig = go.Figure()

# Calculate all approximations
frames = []

for k in range(1, n_terms + 1):
    # Calculate cumulative sum up to k terms
    cumulative = np.zeros_like(x)
    for j in range(1, k + 1):
        term = (4 / np.pi) * np.sin((2 * j - 1) * x) / (2 * j - 1)
        cumulative += term

    # Create frame for this approximation
    frames.append(
        go.Frame(
            data=[
                go.Scatter(
                    x=x,
                    y=cumulative,
                    mode="lines",
                    line=dict(color="blue", width=2),
                    name=f"{k} term(s)",
                ),
                go.Scatter(
                    x=x,
                    y=np.where(x < 0, -1, 1),
                    mode="lines",
                    line=dict(color="black", dash="dash", width=2),
                    name="Target",
                ),
            ],
            name=str(k),
        )
    )

# Set initial data (1 term)
cumulative_1 = (4 / np.pi) * np.sin(x)
fig.add_trace(
    go.Scatter(
        x=x,
        y=cumulative_1,
        mode="lines",
        line=dict(color="blue", width=2),
        name="1 term",
    )
)
fig.add_trace(
    go.Scatter(
        x=x,
        y=np.where(x < 0, -1, 1),
        mode="lines",
        line=dict(color="black", dash="dash", width=2),
        name="Target",
    )
)

# Add frames to figure
fig.frames = frames

# Add slider
fig.update_layout(
    title="Fourier Series Approximation of a Square Wave",
    xaxis_title="x",
    yaxis_title="f(x)",
    autosize=True,
    margin=dict(b=200),
    sliders=[
        {
            "active": 0,
            "yanchor": "top",
            "y": 0,
            "xanchor": "left",
            "currentvalue": {
                "prefix": "Number of terms: ",
                "visible": True,
                "xanchor": "right",
            },
            "pad": {"b": 10, "t": 10},
            "len": 0.9,
            "x": 0.1,
            "steps": [
                {
                    "args": [
                        [f.name],
                        {
                            "frame": {"duration": 0, "redraw": True},
                            "mode": "immediate",
                            "transition": {"duration": 0},
                        },
                    ],
                    "label": str(k),
                    "method": "animate",
                }
                for k, f in enumerate(frames, 1)
            ],
        }
    ],
)

fig.show(config={'responsive': True})
