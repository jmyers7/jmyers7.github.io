\prob Describe the curve that is parameterized by
	\[x=t \quad y=t^2.
	\]

\bigskip

\bigskip

\prob Describe the curve that is parameterized by
	\[x = -\cos{(t)} \quad y =\sin{(t)}.
	\]

\bigskip
\textcolor{red}{The unit circle, trace clockwise.}
\bigskip

\prob Describe the curve that is parameterized by
	\[x = \cos{t} \quad y = \sin{t} \quad z = t.
	\]

\bigskip
\textcolor{red}{A helix wrapping around the $z$-axis of radius $1$.}
\bigskip

\bigskip
\prob Find a parameterization for the following curves.

\begin{enumerate}
\item A circle of radius 3 centered on the $z$-axis and lying in the plane $z=5$.

\bigskip
\textcolor{red}{
	\[x = 3\cos{t}, \quad y = 3\sin{t}, \quad z=5
	\]}
\bigskip

\item The circle of radius 2 in the $xy$-plane centered at the origin, clockwise.

\bigskip
\textcolor{red}{
	\[x = 2\sin{t}, \quad y = 2\cos{t}, \quad z=0
	\]}
\bigskip

\item The circle of radius 2 in the $xz$-plane centered at the origin.

\bigskip
\textcolor{red}{
	\[x = 2\cos{t}, \quad y = 0, \quad z= 2\sin{t}
	\]}
\bigskip

\item The curve in which the plane $z=2$ cuts the surface $z=\sqrt{x^2+y^2}$.

\bigskip
\textcolor{red}{
	\[x = 2\cos{t}, \quad y= 2\sin{t} , \quad z=2
	\]}
\bigskip
\end{enumerate}
