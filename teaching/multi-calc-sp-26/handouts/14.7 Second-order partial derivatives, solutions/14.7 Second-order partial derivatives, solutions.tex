\prob If $f(x,y) = x^4y^2 - x^3y$, find:

\medskip
\begin{enumerate}
\item $f_x(x,y) = \textcolor{red}{4x^3y^2-3x^2y}$
\item $f_y(x,y) = \textcolor{red}{2x^4y - x^3}$
\item $f_{xx}(x,y) = \textcolor{red}{12x^2y^2-6xy}$
\item $f_{xy}(x,y) = \textcolor{red}{8x^3y-3x^2}$
\item $f_{yx}(x,y) = \textcolor{red}{8x^3y-3x^2}$
\item $f_{yy}(x,y) = \textcolor{red}{2x^4}$
\end{enumerate}
