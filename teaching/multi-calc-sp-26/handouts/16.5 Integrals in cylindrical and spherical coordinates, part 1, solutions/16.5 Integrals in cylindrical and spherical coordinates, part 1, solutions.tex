\prob Evaluate
	\[\int_W \sin{(x^2+y^2)} \ dV
	\]
where $W$ is the solid cylinder with height 4 and with base of radius 1 centered on the $z$-axis at $z=-1$.

\bigskip
\textcolor{red}{
\begin{align*}
\int_0^{2\pi} \int_0^1 \int_{-1}^3 \sin{(r^2)}r \ dz dr d\theta  &= 4 \int_0^{2\pi} \int_0^1 \sin{(r^2)} r \ dr d\theta \\
&= -2 \int_0^{2\pi} ( \cos{(1)} -1 )  \ d\theta \\
&= -4\pi(\cos{(1)} -1 ).
\end{align*}}

\prob Find the volume between the cone $z=\sqrt{x^2+y^2}$ and the plane $z=10+x$ above the disk $x^2+y^2 \leq 1$.

\bigskip
\textcolor{red}{
\begin{align*}
\int_0^{2\pi} \int_0^1 \int_r^{10+r\cos{(\theta)}} r \ dzdrd\theta &= \int_0^{2\pi} \int_0^1 ( 10r+r^2\cos{(\theta)} -r^2 ) \ drd\theta \\
&= \int_0^{2\pi} \left( \frac{14}{3} + \frac{1}{3} \cos{(\theta)} \right) \ d\theta \\
&= \frac{28\pi}{3}.
\end{align*}}
