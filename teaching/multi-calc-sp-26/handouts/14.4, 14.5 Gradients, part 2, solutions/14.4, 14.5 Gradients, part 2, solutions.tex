\prob Compute $f_x(x,y,z)$, $f_y(x,y,z)$, and $f_z(x,y,z)$ for the function
	\[f(x,y,z) = x^2y^2z + x\cos{(y)} + z \sin{(xy)}.
	\]

\bigskip
\textcolor{red}{
\begin{align*}
f_x(x,y,z) &= 2xy^2z + \cos{(y)} + yz\cos{(xy)}, \\
f_y(x,y,z) &= 2x^2yz - x\sin{(y)} + xz\cos{(xy)}, \\
f_z(x,y,z) &= x^2y^2 + \sin{(xy)}.
\end{align*}}
\bigskip

\prob Find the directional derivative of $f (x, y, z) = xy + z$ at the point $(-1, 0, 1)$ in the direction of the vector $\vec{v} = 2 \vec{i} + \vec{k}$.

\bigskip
\textcolor{red}{First we get
	\[\nabla f(x,y,z) = y\vec{i} + x\vec{j} + \vec{k}
	\]
so that $\nabla f(-1,0,1) = -\vec{j}+\vec{k}$. Now we get a unit vector
	\[\vec{u} = \frac{\vec{v}}{||\vec{v}||} = \frac{1}{\sqrt{5}}(2\vec{i} + \vec{k}).
	\]
Thus
	\[f_{\vec{u}}(-1,0,1) = \nabla f(-1,0,1) \cdot \vec{u} = \frac{1}{\sqrt{5}}.
	\]}

\prob Compute the gradient of the function
	\[f(x,y,z) = \frac{yz^2}{1+x^2}.
	\]

\bigskip
\textcolor{red}{
	\[\nabla f(x,y,z) = \frac{-2xyz^2}{(1+x^2)^2} \vec{i} + \frac{z^2}{1+x^2} \vec{j} + \frac{2yz}{1+x^2} \vec{k}.
	\]}
\bigskip

\prob Find the directional derivative of $f(x,y,z) = xy+z^2$ at the point $(2,3,4)$ in the direction of maximum rate of change.

\bigskip
\textcolor{red}{This is just the gradient at $(2,3,4)$:
	\[\nabla f(2,3,4) = 3 \vec{i} +2\vec{j}+ 8 \vec{k}.
	\]}

\prob Find a normal vector to the surface $z=x^2+y^2$ at the point $(-1,1,2)$.

\bigskip
\textcolor{red}{To find a normal vector, we first view the surface as the level surface of the function
	\[w = f(x,y,z) = x^2+y^2-z
	\]
when $w=0$. Its gradient at $(-1,1,2)$ is
	\[\nabla f(-1,1,2) = -2\vec{i} +2\vec{j} - \vec{k},
	\]
and this vector is normal to the surface at $(-1,1,2)$.}
\bigskip

\prob Find the tangent plane to the surface $z=x^2+y^2$ at the point $(-1,1,2)$.

\bigskip
\textcolor{red}{From the previous problem we already know that
	\[f_x(-1,1,2) = -2, \quad f_y(-1,1,2) = 2, \quad f_z(-1,1,2) = -1.
	\]
Hence the tangent plane is
	\[-2(x+1) + 2(y-1) - (z-2) = 0.
	\]}
