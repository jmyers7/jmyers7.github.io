\bigskip
\prob Find the volume of the region bounded by the planes $z=3y$, $z=y$, $y=1$, $x=1$, and $x=2$.

\bigskip
\textcolor{red}{
	\[\int_1^2 \int_0^1 \int_y^{3y} \ dz\ dy \ dx = 1.
	\]}
\bigskip

\prob Find the volume of the pyramid with base in the plane $z=-6$ and sides formed by the three planes $y=0$ and $y-x=4$ and $2x+y+z=4$.

\bigskip
\textcolor{red}{
	\[\int_0^6 \int_{y-4}^{\frac{10-y}{2}} \int_{-6}^{4-2x-y} \ dz \ dx \ dy = 162.
	\]}
\bigskip


\prob Find the mass of a triangular-shaped solid bounded by the planes $z=1+x$, $z=1-x$, $z=0$, and with $0 \leq y \leq 3$. The density is $\delta(x,y,z)=10-z$ in units of grams per cubic centimeters, and $x,y,z$ are in centimeters.

\bigskip
\textcolor{red}{
	\[\int_0^1 \int_0^3 \int_{z-1}^{1-z} ( 10-z) \ dx \ dy \ dz = 29 \ \tn{grams}
	\]}
