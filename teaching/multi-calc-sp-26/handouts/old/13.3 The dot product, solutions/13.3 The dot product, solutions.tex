\prob Compute the angle between the vectors
	\[\vec{i} + \vec{j} + \vec{k} \quad \tn{and} \quad \vec{i} -\vec{j} - \vec{k}.
	\]

\bigskip
\textcolor{red}{Let $\vec{v}$ be the first vector and $\vec{w}$ the second. We then have
	\[\vec{v} \cdot \vec{w} = (\vec{i} + \vec{j} + \vec{k}) \cdot (\vec{i} -\vec{j} - \vec{k}) = 1 -1 -1 = -1
	\]
and
	\[||\vec{v}|| = ||\vec{w}|| = \sqrt{3},
	\]
so that
	\[\vec{v} \cdot \vec{w} = ||\vec{v}|| ||\vec{w}|| \cos{\theta} \quad \Rightarrow \quad -1 = \sqrt{3} \cdot \sqrt{3} \cos{\theta}.
	\]
Then $\theta \approx 1.91$, which is about $109^\circ$.}

\prob Give a unit vector which is perpendicular to $\vec{v} = 2\vec{i} +3\vec{j}$.

\bigskip
\textcolor{red}{ First we get vector $\vec{w} = w_1 \vec{i} + w_2 \vec{j}$ which is perpendicular to $\vec{v}$. Since we muat have $\vec{v} \cdot \vec{w}=0$, we have
	\[2w_1 + 3w_2=0,
	\]
which shows we can pick $\vec{w} = -3\vec{i} + 2\vec{j}$. Our answer is then
	\[\frac{\vec{w}}{||\vec{w}||} = \frac{1}{\sqrt{13}} (-3\vec{i} + 2\vec{j}).
	\]}
