\bigskip
\prob Evaluate
	\[\int_W \sin{(\phi)} \ dV
	\]
where $W$ is the region $0 \leq \theta \leq 2 \pi$, $0 \leq \phi \leq \pi/4$, and $1 \leq \rho \leq 2$.

\bigskip
\textcolor{red}{
\begin{align*}
\int_0^{2\pi} \int_0^{\pi/4} \int_1^2 \sin^2{(\phi)}\rho^2 \ d\rho d\phi d \theta &= \frac{7}{3} \int_0^{2\pi} \int_0^{\pi/4} \sin^2{(\phi)} \ d\phi d\theta \\
&= \frac{7}{3} \int_0^{2\pi} \int_0^{\pi/4} \left( \frac{1}{2} - \frac{1}{2} \cos{(2\phi)} \right) \ d\phi d\theta \\
&= \frac{7}{3}\left( \frac{\pi}{8} - \frac{1}{4}\right) \int_0^{2\pi}  d\theta \\
&= \frac{14\pi}{3}\left( \frac{\pi}{8} - \frac{1}{4}\right).
\end{align*}}

\bigskip

\prob Evaluate
	\[\int_W (x^2+y^2) \ dV,
	\]
where $W$ lies between the spheres $x^2+y^2+z^2=4$ and $x^2+y^2+z^2=9$.

\bigskip
\textcolor{red}{
\begin{align*}
\int_W (x^2+y^2) \ dV &= \int_0^\pi \int_0^{2\pi} \int_2^3 (\rho^2 \sin^2{(\phi)}) \rho^2 \sin{(\phi)} \ d\rho d\theta d\phi \\
&= \frac{211}{5}\int_0^\pi \int_0^{2\pi} \sin^3{(\phi)} \ d\theta d\phi \\
&=  \frac{422\pi}{5} \int_0^\pi \sin^3{(\phi)} \ d\phi \\
&= \frac{422\pi}{5}\int_0^\pi(\sin{(\phi)} - \sin{(\phi)} \cos^2{(\phi)}) \ d\phi \\
&= \frac{1688\pi}{15}
\end{align*}}
