\prob Compute the cross product
	\[(2\vec{i}+\vec{j}) \times (3\vec{i}+4\vec{j}+5\vec{k})
	\]

\bigskip
\textcolor{red}{
	\[
\begin{vmatrix} \vec{i} & \vec{j} & \vec{k} \\ 2 & 1 & 0 \\ 3 & 4 & 5 \end{vmatrix} = (1\cdot 5-4\cdot 0)\vec{i}-(2\cdot 5+3\cdot 0)\vec{j}+(2\cdot 4-3\cdot 1)\vec{k} =5\vec{i}-10\vec{j}+5\vec{k}.
	\]}	
	

\bigskip
\prob Compute the cross product
	\[\ang{1,2,3} \times \ang{0,4,-2}.
	\]
	
\bigskip
\textcolor{red}{
	\[
\begin{vmatrix} \vec{i} & \vec{j} & \vec{k} \\ 1 & 2 & 3 \\ 0 & 4 & -2 \end{vmatrix} = \ang{2(-2)-4\cdot 3,-(1\cdot (-2) - 0 \cdot 3), 1\cdot 4 - 0 \cdot 2} = \ang{-16,2,4}.
	\]}

\bigskip
\prob Find the area of the parallelogram with vertices
	\[A = (1,-4,3), \quad B=(2,-2,6), \quad C=(1,0,1), \quad D= ( 2,2,4).
	\]

\bigskip
\textcolor{red}{The parallelogram is spanned by the vectors
	\[\overrightarrow{AB} = \vec{i}+2\vec{j} + 3\vec{k} \quad \tn{and} \quad \overrightarrow{AC} = 4\vec{j} -2\vec{k}.
	\]
Thus its area is the magnitude of the cross product $(\vec{i} + 2\vec{j} + 3\vec{k}) \times (4\vec{j} -2 \vec{k})$. But we found (in the previous problem) that this cross product is $-16\vec{i} +2\vec{j} +4\vec{k}$. Thus the area we want is
	\[||-16\vec{i} +2\vec{j} +4\vec{k}|| = \sqrt{16^2 + 2^2 + 4^2} = 2\sqrt{69}.
	\]}

\bigskip
\prob Compute $\vec{i} \times \vec{j}$, $\vec{j} \times \vec{i}$, and $\vec{j} \times \vec{k}$.

\bigskip
\textcolor{red}{
	\[\vec{i} \times \vec{j} = \vec{k}, \quad \vec{j} \times \vec{i} = -\vec{k}, \quad \vec{j} \times \vec{k} = \vec{i}.
	\]}
\bigskip

\prob Find an equation for the plane through the three points
	\[(1,2,3), \ (1,0,-2), \ (2,3,-1).
	\]

\bigskip
\textcolor{red}{Let $P,Q,R$ be the three points, in the order written. We then have
	\[\overrightarrow{PQ} = -2\vec{j}-5\vec{k}, \quad \overrightarrow{PR} = \vec{i} +\vec{j} -4\vec{k}.
	\]
A normal vector to the plane is then
	\[(-2\vec{j}-5\vec{k}) \times (\vec{i} +\vec{j} -4\vec{k}) = 13\vec{i} -5\vec{j} +2\vec{k}.
	\]
Hence an equation is
	\[13(x-1) -5(y-2)+2(z-3) =0.
	\]}
