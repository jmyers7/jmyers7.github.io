\prob Compute the gradient of the function
	\[f(x,y) = y \ln{(x)} +xy^2
	\]
at the point $(1,2)$.

\medskip
\textcolor{red}{We have
	\[\nabla f(x,y) = (y/x+y^2)\vec{i}+(\ln{(x)}+2xy)\vec{j},
	\]
so
	\[\nabla f(1,2) = 6\vec{i} +4\vec{j}.
	\]}
\medskip

\prob Compute the gradient of the function
	\[f(x,y)=\ln{(x^2 +xy)}
	\]
at the point $(4,1)$.

\medskip
\textcolor{red}{We have
	\[\nabla f(x,y) = \left(\frac{2x+y}{x^2+xy}\right)\vec{i}  + \left( \frac{x}{x^2+xy}\right) \vec{j}
	\]
so
	\[\nabla f(4,1) = \left(\frac{9}{20}\right) \vec{i} + \left( \frac{1}{5} \right) \vec{j}.
	\]}
\medskip

\prob If $f(x,y) = xy+y^3$ and $\vec{u} = \frac{1}{5}(3\vec{i}-4\vec{j}) $, find $f_{\vec{u}}(1,2)$.

\medskip
\textcolor{red}{We have
	\[f_{\vec{u}}(x,y) = \nabla f(x,y) \cdot \vec{u} = \frac{1}{5} (y\vec{i}+(x+3y^2)\vec{j}) \cdot (3\vec{i}-4\vec{j}) = \frac{1}{5} \left( 3y-4x-12y^2 \right).
	\]
Hence
	\[f_{\vec{u}}(1,2) = -\frac{46}{5}.
	\]}

\prob If $f(x,y) = 3x^2 - 2y^2$, find the instantaneous rate of change of $f$ at $(-3/4,0)$ in the direction from $(-3/4,0)$ to $(0,1)$.

\medskip
\textcolor{red}{First get a vector in the correct direction:
	\[(3/4)\vec{i} + \vec{j}.
	\]
Now normalize it to get the unit vector
	\[\vec{u} = \frac{4}{5} \left((3/4)\vec{i} + \vec{j}\right).
	\]
Then
	\[f_{\vec{u}}(x,y) = \nabla f(x,y) \cdot \vec{u} = \frac{4}{5} (6x\vec{i}-4y\vec{j}) \cdot ((3/4)\vec{i}+\vec{j}) = \frac{4}{5}\left( \frac{9}{2}x-4y\right).
	\]
Hence
	\[f_{\vec{u}}(-3/4,0) = -\frac{27}{10}.
	\]}
