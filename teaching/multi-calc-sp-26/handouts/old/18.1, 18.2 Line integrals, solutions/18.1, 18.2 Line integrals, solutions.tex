\prob Consider the vector field $\vec{F}$ shown below together with the paths $C_1$, $C_2$, and $C_3$. Arrange the line integrals over the curves in ascending order.

\bigskip
\includegraphics[scale=1.5]{pic7.pdf}

\bigskip
\textcolor{red}{
	\[\int_{C_3} \vec{F} \cdot d\vec{r} < \int_{C_1} \vec{F} \cdot d\vec{r} < \int_{C_2} \vec{F} \cdot d \vec{r}.
	\]}


\bigskip
\prob Compute the line integrals.

\bigskip
\begin{enumerate}
\item $\displaystyle\int_C \ang{x+y,y} \cdot d \vec{r}$ where $C$ is the quarter unit circle from $(1,0)$ to $(0,1)$.

\bigskip
\textcolor{red}{A parameterization is
	\[\vec{r}(t) = \ang{\cos{(t)},\sin{(t)}}, \quad t\in [0,\pi/2].
	\]
So
\begin{align*}
\int_C \vec{F} \cdot d\vec{r} &= \int_0^{\pi/2} \vec{F}(\vec{r}(t)) \cdot \vec{r}\ '(t) \ d t \\
&= \int_{0}^{\pi/2} \ang{\cos{(t)}+\sin{(t)},\sin{(t)}} \cdot \ang{-\sin{(t)},\cos{(t)}} \ dt \\
&= - \int_0^{\pi/2} \sin^2{(t)} \ dt \\
&= - \int_0^{\pi/2} \left( \frac{1}{2} - \frac{1}{2} \cos{(2t)} \right) \ dt \\
&= - \frac{\pi}{4}.
\end{align*}}
\bigskip

\item $\displaystyle\int_C \ang{3, y+5} \cdot d\vec{r}$ where $C$ is the line from $(0,0)$ to $(0,3)$.

\bigskip
\textcolor{red}{A parameterization is
	\[\vec{r}(t) = \ang{0,t}, \quad t\in [0,3].
	\]
So
\begin{align*}
\int_C \ang{3, y+5} \cdot d\vec{r} &= \int_0^3 \ang{3,t+5} \cdot \ang{0,1} \ dt \\
&= \int_0^3 (t+5) \ dt \\
&= \frac{39}{2}.
\end{align*}}
\bigskip

\item $\displaystyle\int_C \ang{x^2+y, y^3 ,0} \cdot d\vec{r}$ where $C$ consists of the two line segments from $(4,0,0)$ to $(4,3,0)$ to $(0,3,0)$.

\bigskip
\textcolor{red}{We need to parameterize the two lines, which we call $C_1$ and $C_2$. They are
	\[\vec{r}_1(t) = \ang{4,3t,0}, \quad t\in [0,1]
	\]
and
	\[\vec{r}_2(t) = \ang{4-4t,3,0}, \quad t\in [0,1].
	\]
So
\begin{align*}
\int_C \ang{x^2+y, y^3 ,0} \cdot d\vec{r} &= \int_{C_1} \ang{x^2+y, y^3 ,0} \cdot d\vec{r} + \int_{C_2}\ang{x^2+y, y^3 ,0} \cdot d\vec{r} \\
&= \int_0^1 \ang{16+3t, 27t^3 ,0} \cdot \ang{0,3,0} \ dt + \int_ 0^1\ang{(4-4t)^2+3, 27 ,0} \cdot \ang{-4,0,0} \ dt \\
&= 81 \int_0^1 t^3 \ dt -4 \int_0^1 ( (4-4t)^2 + 3) \ dt \\
&=\frac{81}{4} - \frac{100}{3} \\
&= - \frac{157}{12}.
\end{align*}}
\bigskip

\item $\displaystyle\int_C (\sin{(x)} \vec{i} + \cos{(x^2)} \vec{j}) \cdot d\vec{r}$ where $C$ is the parabola $y=x^2$ from $(0,0)$ to $(2,4)$.

\bigskip
\textcolor{red}{A parameterization is
	\[\vec{r}(t) = \ang{ t,t^2} , \quad t\in [ 0,2].
	\]
So
\begin{align*}
\int_C (\sin{(x)} \vec{i} + \cos{(x^2)} \vec{j}) \cdot d\vec{r} &= \int_0^2 ( \sin{(t)} \vec{i} + \cos{(t^2)} \vec{j} ) \cdot ( \vec{i} +2t \vec{j} ) \ dt \\
&= \int_0^2 (\sin{(t)} + 2t \cos{(t^2)} ) \ dt \\
&= -\cos{(2)} + \sin{(4)} + 1.
\end{align*}}

\end{enumerate}

\bigskip
\prob Compute the line integrals.

\bigskip
\begin{enumerate}
\item $\displaystyle\int_C 3y \ dx + 4x \ dy$ where $C$ is the line from $(1,3)$ to $(5,9)$.

\bigskip
\textcolor{red}{A parameterization is
	\[\vec{r}(t) = \ang{1+4t,3+ 6t}, \quad t\in [0,1].
	\]
So
\begin{align*}
\int_C 3y \ dx + 4x \ dy &= \int_0^1 ( 12(3+6t) + 24(1+4t) ) \ dt \\
&= 144.
\end{align*}}
\bigskip

\item $\displaystyle\int_C x \ dx + z \ dy - y \ dz$ where $C$ is the circle of radius $3$ in the $yz$-plane centered at the origin, oriented counterclockwise when viewed from the positive $x$-axis.

\bigskip
\textcolor{red}{A parameterization is
	\[\vec{r}(t) = \ang{0,3\cos{(t)},3\sin{(t)}}, \quad t\in [0,2\pi].
	\]
So
\begin{align*}
\int_C x \ dx + z \ dy - y \ dz &= \int_0^{2\pi} ( -9\sin^2{(t)} -9\cos^2{(t)}  ) \ dt \\
&= -9 \int_0^{2\pi} dt \\
&= -18\pi.
\end{align*}}

\end{enumerate}
