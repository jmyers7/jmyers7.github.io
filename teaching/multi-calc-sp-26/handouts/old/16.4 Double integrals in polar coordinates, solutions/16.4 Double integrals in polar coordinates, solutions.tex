\prob Evaluate
	\[\int_R \sqrt{x^2+y^2} \ dA
	\]
where $R$ is $4 \leq x^2+y^2 \leq 9$.

\bigskip
\textcolor{red}{
	\[\int_R \sqrt{x^2+y^2} \ dA = \int_2^3 \int_0^{2\pi} r^2 \ d\theta dr = 2\pi \int_2^3 r^2 \ dr = \frac{38\pi}{3}.
	\]}
\bigskip

\prob Evaluate
	\[\int_R \sin{(x^2+y^2)} \ dA
	\]
where $R$ is the disk of radius $2$ centered at the origin.

\bigskip
\textcolor{red}{
	\[\int_R \sin{(x^2+y^2)} \ dA = \int_0^2 \int_0^{2\pi} r\sin{(r^2)} \ d\theta dr = 2\pi \int_0^2 r\sin{(r^2)} \ dr = -\pi \cos{(r^2)}\big|_0^2 = -\pi \cos{(4)} +\pi.
	\]}
\bigskip

\prob Convert the iterated integral to polar coordinates and evaluate
	\[\int_{-1}^0 \int_{-\sqrt{1-x^2}}^{\sqrt{1-x^2}} x \ dy \ dx.
	\]

\bigskip
\textcolor{red}{
\begin{align*}
\int_{-1}^0 \int_{-\sqrt{1-x^2}}^{\sqrt{1-x^2}} x \ dy \ dx &= \int_{\pi/2}^{3\pi/2} \int_0^1 r^2 \cos{(\theta)} \ drd\theta \\
&= \frac{1}{3} \int_{\pi/2}^{3\pi/2} \cos{(\theta)} \ d\theta \\
&= \frac{1}{3}( \sin{(3\pi/2)} - \sin{(\pi/2)}) \\
&= -\frac{2}{3}. 
\end{align*}}
