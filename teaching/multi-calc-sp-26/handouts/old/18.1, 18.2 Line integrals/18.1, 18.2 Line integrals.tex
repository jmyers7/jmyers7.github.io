\prob Compute the line integrals.

\medskip
\begin{enumerate}
\item $\displaystyle\int_C \ang{x+y,y} \cdot d \vec{r}$ where $C$ is the quarter unit circle from $(1,0)$ to $(0,1)$.\vfill
\item $\displaystyle\int_C \ang{3, y+5} \cdot d\vec{r}$ where $C$ is the line from $(0,0)$ to $(0,3)$.\vfill\newpage
\item $\displaystyle\int_C \ang{x^2+y, y^3 ,0} \cdot d\vec{r}$ where $C$ consists of the two line segments from $(4,0,0)$ to $(4,3,0)$ to $(0,3,0)$.\vfill
\item $\displaystyle\int_C (\sin{(x)} \vec{i} + \cos{(x^2)} \vec{j}) \cdot d\vec{r}$ where $C$ is the parabola $y=x^2$ from $(0,0)$ to $(2,4)$.\vfill
\end{enumerate}

\prob Compute the line integrals.

\medskip
\begin{enumerate}
\item $\displaystyle\int_C 3y \ dx + 4x \ dy$ where $C$ is the line from $(1,3)$ to $(5,9)$.\vfill
\item $\displaystyle\int_C x \ dx + z \ dy - y \ dz$ where $C$ is the circle of radius $3$ in the $yz$-plane centered at the origin, oriented counterclockwise when viewed from the positive $y$-axis.\vfill
\end{enumerate}
