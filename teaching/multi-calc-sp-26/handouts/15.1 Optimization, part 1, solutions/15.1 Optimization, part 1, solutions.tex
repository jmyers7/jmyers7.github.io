\prob Find all critical points of $f(x,y) = x^2+y^2$. Classify the points as maximums, minimums, or neither.

\bigskip
\textcolor{red}{The equation
	\[\nabla f(x,y) = \ang{2x,2y} = \ang{0,0}
	\]
implies $x=y=0$. Thus $(0,0)$ is a critical point of $f$. The point $(0,0,0)$ is a local minimum, which can be verified by checking the graph.}
\bigskip

\prob Find all critical points of $f(x,y) = -\sqrt{x^2+y^2}$. Classify the points as maximums, minimums, or neither.

\bigskip
\textcolor{red}{The gradient
	\[\nabla f(x,y) = \left\langle -\frac{x}{\sqrt{x^2+y^2}} , -\frac{y}{\sqrt{x^2+y^2}} \right\rangle
	\]
is undefined at $(0,0)$, so this point is a critical point. There is a local maximum at $(0,0,0)$, which can be verified by checking the graph.}
\bigskip

\prob Find all critical points of $f(x,y) = x^2-y^2$. Classify the points as maximums, minimums, or neither.

\bigskip
\textcolor{red}{The equation
	\[\nabla f(x,y) = \ang{2x,-2y} = \ang{0,0}
	\]
implies $x=y=0$, so $(0,0)$ is a critical point. There is neither a local maximum nor local minimum at $(0,0,0)$, which means it's a saddle point.}

\prob Find all critical points of $f(x,y) = 5+6x-x^2+xy-y^2$. Classify the points as maximums, minimums, or neither.

\bigskip
\textcolor{red}{We first find critical points:
	\[\nabla f(x,y) = \ang{6-2x+y, x-2y} = \ang{0,0}
	\]
implies $x = 4$ and $y=2$. Then
	\[D = \begin{vmatrix}-2 & 1 \\ 1 & -2 \end{vmatrix} = 4  -1 = 3 >0.
	\]
Since $f_{xx}(4,2) = -2 < 0$, we see $(4,2,17)$ is a local maximum.}
\bigskip

\prob Find all critical points of $f(x,y) = x^2y+2y^2-2xy+6$. Classify the points as maximums, minimums, or neither.

\bigskip
\textcolor{red}{We first find critical points:
	\[\nabla f(x,y) = \ang{ 2xy - 2y, x^2+4y-2x} = \ang{0,0}
	\]
implies either $x=y=0$, $x=2$ and $y=0$, or $x=1$ and $y=1/4$. We have
	\[D = \begin{vmatrix} 2y & 2x-2 \\ 2x-2 & 4 \end{vmatrix} = 8y - (2x-2)^2
	\]
Plugging in $(0,0)$ gives $D = -4 < 0$, so $(0,0,6)$ is a saddle point. Plugging in $(2,0)$ gives $D=-4<0$ again, so $(2,0,6)$ is also a saddle point. Plugging in $(1,1/4)$ gives $D=2>0$. Since also $f_{xx}(1,1/4)=1/2 >0$, this means there is a local minimum at $(1,1/4,47/8)$.}
