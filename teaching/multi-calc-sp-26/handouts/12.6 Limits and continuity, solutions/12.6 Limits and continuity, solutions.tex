\prob Show that the limit
	\[\lim_{(x,y) \to (0,0)} \frac{4x^2}{x^2+y^2}
	\]
does not exist.

\bigskip
\textcolor{red}{We pick two paths toward the origin in the $xy$-plane, and show that the limits do not match along these paths. First, we go along the path $x=0$:
	\[\lim_{(x,y) \to (0,0)} \frac{4x^2}{x^2+y^2} = \lim_{y \to 0} \frac{4\cdot0^2}{0^2+y^2} = 0.
	\]
Then we go along the path $y=0$:
	\[\lim_{(x,y) \to (0,0)} \frac{4x^2}{x^2+y^2} = \lim_{x \to 0} \frac{4x^2}{x^2+0^2} = 4.
	\]
Since $4\neq 0$, the multivariable limit does not exist.}
\bigskip

\prob Show that the limit
	\[\lim_{(x,y)\to (0,0)} \frac{ xy^2}{x^2+y^4}
	\]
does not exist.

\bigskip
\textcolor{red}{Go toward $(0,0)$ along the line $x=0$:
	\[\lim_{(x,y)\to(0,0)} \frac{xy^2}{x^2+y^4} = \lim_{y\to 0} \frac{0}{y^4} = 0.
	\]
Now go toward $(0,0)$ along the line $x=y^2$:
	\[\lim_{(x,y)\to(0,0)} \frac{xy^2}{x^2+y^4} = \lim_{y\to 0} \frac{y^4}{2y^4} = 1/2.
	\]
These two limits don't agree, so the $2$-variable limit doesn't exist.}

\bigskip
\prob Compute
	\[\lim_{(x,y) \to (2,-1)} \frac{x^2y+xy^2}{x^2-y^2}
	\]
if it exists, or show that it does not exist.

\bigskip
\textcolor{red}{The function is continuous at $(2,-1)$ (it's a quotient of polynomials), so we have
	\[\lim_{(x,y) \to (2,-1)} \frac{x^2y+xy^2}{x^2-y^2} = \frac{2^2(-1)+2(-1)^2}{2^2-(-1)^2} = -\frac{2}{3}.
	\]}
\bigskip

\prob Compute
	\[\lim_{(x,y)\to(0,0)} \frac{x+y}{x-y}
	\]
if it exists, or show that it does not exist.

\bigskip
\textcolor{red}{The limit does not exist. Go toward $(0,0)$ along the line $y=3x$:
	\[\lim_{(x,y)\to(0,0)} \frac{x+y}{x-y} = \lim_{x\to 0} \frac{x+3x}{x-3x} = \lim_{x\to 0} -2 = -2.
	\]
Now go toward $(0,0)$ along the line $y=4x$:
	\[\lim_{(x,y)\to(0,0)} \frac{x+y}{x-y} = \lim_{x\to 0} \frac{x+4x}{x-4x} = \lim_{x\to 0} -5/3 = -5/3.
	\]
These two limits don't agree, so the $2$-variable limit doesn't exist.}

\bigskip
\prob Determine whether or not there is a value of $c$ for which the function
	\[f(x,y) = \begin{cases}
	c+y & : x\leq 3, \\
	5-x+y & : x>3
	\end{cases}
	\]
is continuous. If so, find it. If not, explain why not.

\bigskip
\textcolor{red}{Such a $c$ does exist. When $x\leq 3$, the graph of $f$ coincides with the plane $z = c+y$, but if $x>3$, then the graph of $f$ coincides with the plane $z=5-x+y$. Thus $f$ is continuous everywhere, except possibly where the two planes $z=c+y$ and $z=5-x+y$ meet, which occurs when they intersect the vertical plane $x=3$. The intersection of $z=c+y$ with the plane $x=3$ is the line $z=c+y$ in the plane $x=3$, while the intersection of the plane $z=5-x+y$ with the plane $x=3$ is the line $z=2+y$ in the plane $x=3$. This means that if $f$ is going to be continuous, the lines $z=c+y$ and $z=2+y$ must be equal. This occurs when $c=2$.}
