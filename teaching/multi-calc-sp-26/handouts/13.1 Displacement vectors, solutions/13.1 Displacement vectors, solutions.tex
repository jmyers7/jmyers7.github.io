\prob Find a vector with length $2$ which points in the same direction as $\vec{v} = \vec{i} - \vec{j} +2\vec{k}$.

\textcolor{red}{
	\[2\cdot \frac{\vec{v}}{||\vec{v}||} = \frac{2}{\sqrt{1+1+4}}(\vec{i} - \vec{j} +2\vec{k}) = \frac{2}{\sqrt{6}} \vec{i} - \frac{2}{\sqrt{6}}\vec{j} +\frac{4}{\sqrt{6}}\vec{k}
	\]}

\prob Find the values of $a$ making $\vec{v} = 5a\vec{i} -3\vec{j}$ parallel to $\vec{w} = a^2 \vec{i} +6\vec{j}$.

\bigskip
\textcolor{red}{We need to find a number $\lambda$ such that $\lambda \vec{v} = \vec{w}$. This would mean
	\[5\lambda a \vec{i} - 3\lambda \vec{j} = \lambda a^2 \vec{i} +6\lambda \vec{j},
	\]
so that $5\lambda a = a^2$ and $-3\lambda = 6$. The second equation says that $\lambda=-2$, so the first equation becomes $-10a = a^2$. Solving for $a$ gives $a( a+10)=0$, so that $a=0,-10$.}
